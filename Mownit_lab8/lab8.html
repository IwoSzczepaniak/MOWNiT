end

function NewtonBaffler(x)
    if x < 6.0
        fx = 0.75 * ( x - 6.25 )
    elseif x <= 6.50 
        fx = 2.00 * ( x - 6.25 )
    else
        fx = 0.75 * ( x - 6.25 ) 
    end
    return fx
end
